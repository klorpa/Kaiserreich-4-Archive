USA_is_valid_loas_member = {
	country_exists = USA
	NOT = { has_war_with = USA }
	OR = {
		is_ally_with = USA
		AND = {
			is_in_faction = no
			is_subject = no
		}
		is_faction_leader = yes
	}
	has_socialist_government = no
	NOT = { has_government = national_populist }
}

		is_in_faction_with = NEE
	}
}
