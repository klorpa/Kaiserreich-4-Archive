india_claim_ceylon_effect = {
	every_state = {
		limit = {
			OR = {
				is_core_of = CEY
				is_claimed_by = CEY
			}
			NOT = { is_core_of = ROOT }
		}
		add_claim_by = ROOT
		enable_state_integration_80_tooltip = yes
	}
	CEY = { enable_coring_at_eighty_compliance = yes }
	if = {
		limit = {
			var:422.owner = {
				OR = {
					tag = GER
					is_subject_of = GER
				}
				controls_state = 422
			}
		}
		if = {
			limit = { is_in_faction_with = GER }
			custom_effect_tooltip = tooltip_white_line
			GER = { country_event = prf.195 }
			custom_effect_tooltip = if_they_accept_tt
			effect_tooltip = {
				every_state = {
					limit = {
						OR = {
							is_core_of = CEY
							state = 733 #Andaman Islands
							state = 818 #Pondicherry
						}
						OR = {
							is_owned_by = GER
							owner = { is_subject_of = GER }
						}
						is_controlled_by = owner
					}
					transfer_state_to = ROOT
				}
			}
		}
		else_if = {
			limit = {
				GER = {
					has_war = yes
					any_enemy_country = { is_actual_major = yes }
				}
			}
			custom_effect_tooltip = tooltip_white_line
			GER = { country_event = prf.194 }
			effect_tooltip = {
				every_state = {
					limit = {
						OR = {
							is_core_of = CEY
							state = 733 #Andaman Islands
							state = 818 #Pondicherry
						}
						OR = {
							is_owned_by = GER
							owner = { is_subject_of = GER }
						}
						is_controlled_by = owner
					}
					transfer_state_to = ROOT
				}
			}
		}
		else_if = {
			limit = {
				var:422.owner = {
					OR = {
						tag = GER
						is_subject_of = GER
					}
				}
			}
			custom_effect_tooltip = tooltip_white_line
			custom_effect_tooltip = india_will_seize_ceylon_if_germany_at_war
		}
	}
	else_if = {
		limit = { can_declare_war_on = CEY }
		custom_effect_tooltip = tooltip_white_line
		create_wargoal = {
			type = annex_everything
			target = CEY
		}
	}
}
