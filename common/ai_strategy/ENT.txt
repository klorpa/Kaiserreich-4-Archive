ENT_prepare_for_intervention_against_CSA = {
	allowed = {
		is_in_faction_with = ENT
		NOT = { tag = CAN }
		is_american_tag = no
	}

	enable = {
		NOT = { has_global_flag = USA_civil_war_over_flag }
		OR = {
			has_war_with = CSA
			AND = {
				is_in_faction_with = CAN
				check_variable = { CAN.ai_strategy_prepare_for_war@CSA > 0 }
			}
			AND = {
				is_in_faction_with = NEE
				check_variable = { NEE.ai_strategy_prepare_for_war@CSA > 0 }
			}
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = avoid_starting_wars
		value = -400
	}

	ai_strategy = {
		type = force_defend_ally_borders
		id = CAN
		value = 100
	}
	ai_strategy = {
		type = force_defend_ally_borders
		id = NEE
		value = 100
	}

	ai_strategy = {
		type = front_unit_request
		tag = CSA
		value = 60
	}
	ai_strategy = {
		type = prepare_for_war
		id = CSA
		value = 400
	}
	ai_strategy = {
		type = conquer
		id = CSA
		value = 400 #counters avoid_starting_wars
	}
	ai_strategy = {
		type = invade
		id = CSA
		value = -500
	}
}

		id = RAJ
		value = 100
	}

		type = front_unit_request
		tag = HND
		value = 40
	}

		type = prepare_for_war

ENT_buffer_troops_for_invasion_of_france = {
	allowed = {
		is_in_faction_with = ENT
	}
	enable = {
		FRA = {
			has_war_with_major = yes
			has_capitulated = no
		}

		#only if large enough
		num_divisions > 19

		#if we have Spain, don't bother
		NOT = {
			AND = {
				is_in_faction_with = SPA
				SPA = { has_capitulated = no }
			}
		}

		#not if any dangerous neighbours
		surrender_progress < 0.05
		NOT = { any_home_area_neighbor_country = { has_war_with = PREV } }
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = put_unit_buffers
		order_id = 1
		states = {
			459 #Algiers
			805 #Oran
			460 #Constantine
			458 #Tunis
		}
		ratio = 0.2
		area = western_europe
		subtract_invasions_from_need = yes
	}
	ai_strategy = {
		type = put_unit_buffers
		order_id = 1
		states = {
			1 #Corsica
			114 #Sardinia
		}
		ratio = 0.1
		area = western_europe
		subtract_invasions_from_need = yes
	}

	ai_strategy = {
		type = invade
		id = FRA
		value = 100
	}
	ai_strategy = {
		type = invade
		id = SRI
		value = 100
	}

		type = invade
		id = SWF
		value = 100
	}
}

ENT_sync_invasions = {
	allowed = {
		is_in_faction_with = ENT
	}
	enable = {
		has_war_with = INT
		OR = {
			FRA = {
				exists = yes
				has_capitulated = no
			}
			ENG = {
				exists = yes
				has_capitulated = no
			}
		}

		#activate every 60 days
		set_temp_variable = { days = global.num_days }
		modulo_temp_variable = { days = 90 }
		check_variable = { days < 60 }
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_control
		area = western_europe
		ordertype = invasion
		execute_order = no
	}
}
